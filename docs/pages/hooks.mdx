import actions from "../hooks/actions.json";
import filters from "../hooks/filters.json";
import { HookParams } from "../functions.js";

export const HookTOCItem = ({ hook }) => {
	return (
		<li>
			<a href={`#${hook.name}`}>{hook.name}</a>
		</li>
	);
};

export const Hook = ({ hook }) => {
	return (
		<div>
			<h4 id={`${hook.name}`}>
				<code>{hook.name}</code>
			</h4>
			<p>{hook.doc.description}</p>
			<div className="mt-1" dangerouslySetInnerHTML={{ "__html": hook.doc.long_description_html }}></div>
			<HookParams hook={hook} />
		</div>
	);
};
